# GitHub occasionally returns HTTP 500s during git clone. Retry with backoff
# so transient failures don't abort the whole build.
git_clone_retry() {
    local url target attempt
    for arg in "$@"; do
        case "$arg" in
            http*|git@*|ssh://*|git://*) url="$arg" ;;
            -*) ;;
            *) [[ -n "$url" ]] && target="$arg" ;;
        esac
    done
    if [[ -z "$target" && -n "$url" ]]; then
        target="${url##*/}"
        target="${target%.git}"
    fi
    for attempt in 1 2 3 4 5; do
        if git clone "$@"; then
            return 0
        fi
        [[ -n "$target" && -d "$target" ]] && rm -rf "$target"
        local delay=$((attempt * 5))
        echo "git clone failed (attempt $attempt); retrying in ${delay}s..." >&2
        sleep "$delay"
    done
    echo "git clone failed after 5 attempts" >&2
    return 1
}

# Build SDL2 2.32 from source so RSDKv4 links against a modern SDL2 at build
# time. The built libSDL2 is NOT shipped; devices provide libSDL2-2.0.so.0
# at runtime, often with device-specific patches. Shipping ours via
# LD_LIBRARY_PATH would override those patches.
git_clone_retry https://github.com/libsdl-org/SDL.git
cd SDL
git checkout release-2.32.0
mkdir -p build-sonic2013 && cd build-sonic2013
cmake .. -DBUILD_SHARED_LIBS=ON
make -j$(nproc)
make install
cd ../..

#### RSDKv4 (Sonic 1/2 2013 Decompilation) ####

git_clone_retry --recursive https://github.com/RSDKModding/RSDKv4-Decompilation.git
cd RSDKv4-Decompilation

# Check out the latest release tag, or branch HEAD if FORCE_HEAD is set.
# RSDKv4's default branch is `main` and it has proper release tags (1.3.0+).
git fetch --tags
if [[ "${FORCE_HEAD:-false}" == "true" ]]; then
  echo "FORCE_HEAD=true — building from branch HEAD (nightly)"
else
  git checkout $(git describe --tags `git rev-list --tags --max-count=1`)
fi

# Patch upstream bug: RSDKv4/Debug.cpp calls `sprintf(buffer, "%s\n", buffer)`
# which appends '\n\0' past the end of a 256-byte buffer when a log message
# fills it. Harmless on older glibc without -fstack-protector-strong (which
# is why Debian buster builds ran fine), but modern focal GCC enables stack
# protection by default and aborts with "stack smashing detected" on any
# long log line that the engine logs at startup. Apply a tiny inline patch.
patch -p1 <<'PATCH'
--- a/RSDKv4/Debug.cpp
+++ b/RSDKv4/Debug.cpp
@@ -18,11 +18,14 @@
         vsprintf(buffer, msg, args);
         if (endLine) {
             printf("%s\n", buffer);
-            sprintf(buffer, "%s\n", buffer);
+            size_t _L = strlen(buffer);
+            if (_L < sizeof(buffer) - 1) {
+                buffer[_L] = '\n';
+                buffer[_L + 1] = 0;
+            }
         }
         else {
             printf("%s", buffer);
-            sprintf(buffer, "%s", buffer);
         }

         if (engineDebugMode) {
PATCH
# Sanity: no `sprintf(buffer, ..., buffer)` should remain.
if grep -E 'sprintf\(buffer,[^,]+,[[:space:]]*buffer\)' RSDKv4/Debug.cpp; then
    echo "ERROR: sprintf-into-self pattern still present"; exit 1
fi
echo "Debug.cpp patch applied"

# Per RSDKv4/RetroEngine.hpp the RETRO_REVxx macros are ">=" checks, so a
# REV=3 build includes all earlier revisions' code paths. At runtime the
# engine detects which data layout it was given (see Engine.usingOrigins in
# RetroEngine.cpp, which flips based on a data-side variable). That means
# REV=3 is the universal binary: it accepts Sega Forever / Sonic 1-2 (2013)
# mobile Data.rsdk AND Sonic Origins Data.rsdk (after the one-byte signature
# fix from 'RSDKv4' to 'RSDKvB' — Origins labels the same format differently).
# RETRO_USE_HW_RENDER=OFF drops GLEW (device doesn't provide libglvnd; see
# the sonic-cd recipe for background). -pthread fixes an "as-needed" linker
# error on theoraplay's pthread_create call on stricter linkers.
cmake -B build \
  -DCMAKE_BUILD_TYPE=Release \
  -DRETRO_REVISION=3 \
  -DRETRO_USE_HW_RENDER=OFF \
  -DRETRO_DISABLE_PLUS=ON \
  -DCMAKE_EXE_LINKER_FLAGS="-pthread"
cmake --build build --config Release -j$(nproc)

# Stage shipped libraries. Filenames MUST match DT_NEEDED entries exactly;
# `cp -L` follows symlinks so the destination is a real file with the right
# name. libSDL2 is NOT shipped (device provides it).
#
# RSDKv4 only directly NEEDs libvorbisfile — unlike RSDKv3 (Sonic CD) which
# also pulls in libogg/libtheoradec/libvorbis for intro video. RSDKv4 has
# no video playback path, so the dep list is minimal.
mkdir -p build/libs
cp -L /usr/lib/aarch64-linux-gnu/libvorbisfile.so.3 ./build/libs/libvorbisfile.so.3

# Verify library coverage. The binary's output name per CMakeLists:106 is
# whatever RETRO_OUTPUT_NAME is set to; default is "RSDKv4".
bin=$(find build -maxdepth 3 -name RSDKv4 -type f | head -n1)
if [[ -z "$bin" ]]; then
    echo "ERROR: built RSDKv4 binary not found under build/"; exit 1
fi
echo "Binary: $bin"
needed=$(readelf -d "$bin" | awk -F'[][]' '/NEEDED/ {print $2}')
echo "Binary NEEDED:"; echo "$needed"
echo "Staged:"; ls build/libs/

staged_libs="libvorbisfile.so.3"
# ld-linux-aarch64.so.1 is the dynamic loader itself — part of glibc, always
# present on aarch64 systems, referenced via DT_NEEDED on some toolchains.
system_libs="libSDL2-2.0.so.0 libGLESv2.so.2 \
             libc.so.6 libm.so.6 libpthread.so.0 libdl.so.2 librt.so.1 \
             libstdc++.so.6 libgcc_s.so.1 \
             ld-linux-aarch64.so.1"

for lib in $staged_libs; do
    if ! echo "$needed" | grep -qx "$lib"; then
        echo "ERROR: staged $lib but binary does not NEED it"; exit 1
    fi
    if [ ! -f "build/libs/$lib" ]; then
        echo "ERROR: $lib missing from build/libs/"; exit 1
    fi
done

while read -r lib; do
    [[ -z "$lib" ]] && continue
    if ! echo " $staged_libs $system_libs " | grep -q " $lib "; then
        echo "ERROR: binary NEEDs $lib but it is neither staged nor on the system allowlist"; exit 1
    fi
done <<< "$needed"

# Save the binary under both names. The same RSDKv4 build ships to sonic.1
# and sonic.2 — their launchers expect per-game filenames (sonic1 / sonic2)
# so the wrong-game binary is silently ignored in each port folder. Costs
# ~500KB of duplication per port but avoids schema gymnastics in the
# registry to rename artifacts per target_dir.
cp "$bin" build/sonic1
cp "$bin" build/sonic2

cd ..
