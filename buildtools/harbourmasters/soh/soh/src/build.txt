set -e

source /root/buildtools/harbourmasters/lib_build.sh

# Garante OpenSSL 1.1 ARM64 no container.
# Essas libs são copiadas manualmente depois, sem passar pelo stage_libs.
apt-get update
apt-get install -y --no-install-recommends libssl1.1

# Seu fork celshade otimizado.
# Usa o branch wind-waker-style-cel-shading por padrão.
export REF="${REF:-wind-waker-style-cel-shading}"

project_clone https://github.com/Botopai50/ShipwrightTEST2.git

build_sdl2
build_sdl2_net

build_libzip
build_json
build_bzip2
build_tinyxml2
build_opus
build_opusfile
build_fmt
build_spdlog -DSPDLOG_FMT_EXTERNAL=ON

project_configure_and_build GenerateSohOtr \
    -DCMAKE_PREFIX_PATH=/usr/local \
    -DUSE_OPENGLES=1 \
    -DBUILD_REMOTE_CONTROL=1 \
    -DCMAKE_CXX_STANDARD=20 \
    -DCMAKE_CXX_STANDARD_REQUIRED=ON \
    -DCMAKE_C_FLAGS="-O3 -DNDEBUG -DPORTMASTER_CELSHADE_OPTIMIZED=1 -fno-math-errno -fno-trapping-math -ffunction-sections -fdata-sections -fno-semantic-interposition" \
    -DCMAKE_CXX_FLAGS="-O3 -DNDEBUG -DPORTMASTER_CELSHADE_OPTIMIZED=1 -DFMT_CONSTEVAL=constexpr -fno-math-errno -fno-trapping-math -ffunction-sections -fdata-sections -fno-semantic-interposition" \
    -DCMAKE_EXE_LINKER_FLAGS="-Wl,--gc-sections -Wl,--as-needed" \
    -DCMAKE_BUILD_TYPE=Release

# Copia somente as libs diretas do ELF.
# NÃO colocar libcrypto/libssl aqui, porque stage_libs falha se o ELF não pedir direto.
stage_libs soh/soh.elf \
    libz.so.1 \
    libogg.so.0 \
    libvorbis.so.0 \
    libvorbisenc.so.2 \
    libvorbisfile.so.3 \
    libpng16.so.16 \
    libbz2.so.1 \
    libzip.so.5 \
    libtinyxml2.so.10 \
    libopus.so.0 \
    libopusfile.so.0 \
    libSDL2_net-2.0.so.0

# Copia OpenSSL 1.1 manualmente para o pacote final.
mkdir -p "$PROJECT_BUILD_DIR/libs"

for lib in libcrypto.so.1.1 libssl.so.1.1; do
    SRC="/usr/lib/aarch64-linux-gnu/$lib"

    if [ ! -f "$SRC" ]; then
        echo "AVISO: $lib não encontrado em $SRC"
        echo "Procurando no sistema:"
        find /usr /lib -name "$lib" 2>/dev/null || true
    else
        echo "Copiando $SRC para $PROJECT_BUILD_DIR/libs/$lib"
        cp -L "$SRC" "$PROJECT_BUILD_DIR/libs/$lib"
    fi
done

echo "Libs finais empacotadas:"
ls -lh "$PROJECT_BUILD_DIR/libs" || true

echo "Verificando dependências do ELF:"
readelf -d "$PROJECT_BUILD_DIR/soh/soh.elf" | grep NEEDED || true

echo "Build SoH celshade usando Botopai50/ShipwrightTEST2 concluído."
