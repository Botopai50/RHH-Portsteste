## GhostShip

git clone https://github.com/HarbourMasters/Ghostship.git
cd Ghostship

git fetch --tags
git checkout $(git describe --tags `git rev-list --tags --max-count=1`)

# Clone the submodules
git submodule update --init

# Verify where clang is or use the default 'clang' which is v18 on 24.04
if ! command -v clang-18 &> /dev/null; then
    export CC=clang
    export CXX=clang++
else
    export CC=clang-18
    export CXX=clang++-18
fi

mkdir build-ghostship && cd build-ghostship
cmake .. -GNinja \
-DUSE_OPENGLES=1 \
-DBUILD_CROWD_CONTROL=0 \
-DCMAKE_BUILD_TYPE=Release \
-DCMAKE_EXE_LINKER_FLAGS="-ldl -pthread"

cmake --build . -j$(nproc) --config Release --target GeneratePortO2R
cmake --build . -j$(nproc)

# Collate libs
mkdir libs
cp /usr/lib/aarch64-linux-gnu/libogg.so.[0-9]* ./libs/
cp /usr/lib/aarch64-linux-gnu/libvorbis.so.[0-9]* ./libs/
cp /usr/lib/aarch64-linux-gnu/libvorbisenc.so.[0-9]* ./libs/
cp /usr/lib/aarch64-linux-gnu/libvorbisfile.so.[0-9]* ./libs/
cp /usr/lib/aarch64-linux-gnu/libz.so.[0-9]* ./libs/
cp /usr/local/lib/libzip.so.[0-9]* ./libs/
cp /usr/local/lib/libtinyxml2.so.[0-9]* ./libs/
find ./libs/ -name "*.so.[0-9]*.[0-9]*" -delete

cd ../..
