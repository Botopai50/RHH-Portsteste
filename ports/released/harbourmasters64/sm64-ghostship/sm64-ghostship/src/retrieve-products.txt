#!/bin/bash

SRCDIR=$1
DESTDIR=$2

cd "$DESTDIR" || { echo "DESTDIR not found: $DESTDIR"; exit 1; }

cp "$SRCDIR/Ghostship/build-ghostship/Ghostship" "$DESTDIR/"
cp "$SRCDIR/Ghostship/build-ghostship/ghostship.o2r" "$DESTDIR/"
cp "$SRCDIR/Ghostship/build-ghostship/gamecontrollerdb.txt" "$DESTDIR/"

rm -rf ./libs
cp -r "$SRCDIR/Ghostship/build-ghostship/libs" .

mkdir -p "$DESTDIR/tools"

# Copy torch binary
cp "$SRCDIR/Ghostship/build-ghostship/TorchExternal/src/TorchExternal-build/torch" \
   "$DESTDIR/tools/torch"

# Remove old tools content
rm -rf "$DESTDIR/tools/assets"
rm -f "$DESTDIR/tools/config.yml"

# Copy new tools content
cp "$SRCDIR/Ghostship/config.yml" "$DESTDIR/tools/"

# Zip the entire folder as "assets/" inside the zip
pushd "$SRCDIR/Ghostship"
zip -r "$DESTDIR/tools/assets.zip" assets/
popd > /dev/null

echo "Packaging complete. You will also need to update otrgen by hand."
